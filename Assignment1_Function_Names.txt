import random
def roll_dice(sides):
    randomNumber = random.randint(1, sides)
    return randomNumber

def main():
    sides=6
    keep_rolling=True
    while keep_rolling:
        user_input=input("Ready to roll? Enter Q to Quit")
        if user_input.lower() !="q":
            numberOfRolls=roll_dice(sides)
            print("You have rolled a", numberOfRolls)
        else:
            keep_rolling=False

def isValidNumber(userGuess):
    if userGuess.isdigit() and 1<= int(userGuess) <=100:

def getArmstrongValue(number):
    # Initializing Sum and Number of Digits
    sumOfDigits = 0
    totalDigits = 0

    # Calculating Number of individual digits
    temporaryNumber = number
    while temporaryNumber > 0:
        totalDigits = totalDigits + 1
        temporaryNumber = temporaryNumber // 10

    # Finding Armstrong Number
    temporaryNumber = number
    for number in range(1, temporaryNumber + 1):
        remainder = temporaryNumber % 10
        sumOfDigits = sumOfDigits + (remainder ** totalDigits)
        temporyNumber //= 10
    return sumOfDigits
